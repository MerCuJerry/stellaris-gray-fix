#创造灰风总督
create_gray_governor = {
	if = {
		limit = { has_country_flag = gray_warship_active }
		remove_country_flag = gray_warship_active
		event_target:gray_governor = { kill_leader = { show_notification = no } }
		event_target:gray_warship = { delete_fleet = this }#删除灰风母舰
	}
	if = {
		limit = { has_country_flag = gray_army_active }
		remove_country_flag = gray_army_active
		every_owned_army = {
			limit = { army_type = gray_army }
			unassign_leader = this
			remove_army = yes
		}
	}
	set_country_flag = gray_governor_active
	create_leader = {#创建灰风总督
		class = governor
		species = root
		name = NAME_Gray
		gender = female
		event_leader = yes
		skill = 10
		set_age = 0
		skip_background_generation = yes
		custom_catch_phrase = gray_catch_phrase
		custom_description = gray_backstory
		hide_age = yes
		traits = {
			trait = leader_trait_governor_gray
			trait = subclass_governor_economist
			trait = leader_trait_urbanist_3
			trait = leader_trait_scrapper_2
		}
		effect = {
			set_leader_flag = legendary_leader
			set_leader_flag = immune_to_negative_traits
		}
	}
	last_created_leader = {
		save_global_event_target_as = gray_governor
		change_leader_portrait = GrayTempest
		set_leader_flag = has_gained_level_trait
	}
}
create_gray_warship = {
	if = {
		limit = { has_country_flag = gray_governor_active }
		remove_country_flag = gray_governor_active
		event_target:gray_governor = { kill_leader = { show_notification = no } }
	}
	if = {
		limit = { has_country_flag = gray_army_active }
		if = {
			limit = {
				any_owned_army = {
					army_type = gray_army
					exists = planet
				}
			}
			random_owned_army = {
				limit = { army_type = gray_army }
				planet = { save_event_target_as = gray_warship_location }
			}
		}
		else = {
			random_owned_army = {
				limit = { army_type = gray_army }
				fleet = { save_event_target_as = gray_warship_location }
			}
		}
	}
	else = {
		capital_scope = { save_event_target_as = gray_warship_location }
	}
	set_country_flag = gray_warship_active
	create_leader = {
		class = admiral
		species = root
		name = NAME_Gray
		event_leader = yes
		gender = female
		skill = 10
		set_age = 0
		skip_background_generation = yes
		custom_catch_phrase = gray_catch_phrase
		custom_description = gray_backstory
		hide_age = yes
		traits = { 
			trait = leader_trait_admiral_gray
		}
		effect = {
			set_leader_flag = legendary_leader
			set_leader_flag = immune_to_negative_traits
		}
	}
	last_created_leader = {
		save_global_event_target_as = gray_admiral
		change_leader_portrait = GrayTempest
	}
	create_fleet = {
		name = "NAME_Gray"
		settings = {
			can_upgrade = no
			can_change_composition = no
			can_disband = no
			can_change_leader = no
			uses_naval_capacity = no
			spawn_debris = no
		}
		effect = {
			set_owner = root
			create_ship = {
				name = "NAME_Gray"
				design = "NAME_Gray_Warship_Upgrade"
				prefix = no
				upgradable = no					#不可以升级
			}
			save_global_event_target_as = gray_warship
			set_location = event_target:gray_warship_location
			assign_leader = event_target:gray_admiral
			add_modifier = {
				modifier = "gray_warship"
				days = -1
			}
		}
	}
	if = {
		limit = { has_country_flag = gray_army_active }
		remove_country_flag = gray_army_active
		every_owned_army = {
			limit = { army_type = gray_army }
			unassign_leader = this
			remove_army = yes
		}
	}
}
#创造灰风地面单位
create_gray_army = {
	if = {
		limit = { has_country_flag = gray_governor_active }
		remove_country_flag = gray_governor_active
		event_target:gray_governor = {
			kill_leader = { show_notification = no }
		}
	}
	if = {
		limit = { has_country_flag = gray_warship_active }
		event_target:gray_warship = { save_event_target_as = gray_transport_location }
	}
	else = {
		capital_scope = { save_event_target_as = gray_transport_location }
	}
	set_country_flag = gray_army_active
	create_fleet = {
		name = "NAME_Gray"
		effect = {
			set_owner = root
			create_army_transport = {
				ship_name = "NAME_Gray"
				#graphical_culture = "root"
				army_name = "NAME_Gray"
				army_type = "gray_army"
			}
			set_location = {
				target = event_target:gray_transport_location
				distance = 0
				angle = random
			}
		}
	}
	if = {
		limit = { has_country_flag = gray_warship_active }
		remove_country_flag = gray_warship_active
		event_target:gray_admiral = { kill_leader = { show_notification = no } }
		event_target:gray_warship = { delete_fleet = this }
	}
}
