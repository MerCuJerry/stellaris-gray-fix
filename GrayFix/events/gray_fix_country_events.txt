country_event = {
	id = gray_fix.101
	title = "graygoo.500.title"
	desc = "graygoo.501.a.desc"
	desc = "graygoo.501.b.desc"
	desc = "graygoo.501.c.desc"
	desc = "graygoo.501.d.desc"

	diplomatic = yes

	picture = GFX_evt_busy_spaceport
	picture_event_data = {
		portrait = event_target:gray_country.ruler
		room = root
	}

	is_triggered_only = yes

	trigger = {
		from = { is_country_type = gray_fix_country }
		is_ai = no
		has_country_flag = gray_governor_active
		NOT = { has_country_flag = gray_reforming }
		NOT = { has_country_flag = gray_diplomacy_engaged }
		any_owned_leader = { is_same_value = event_target:gray_governor@root }
	}

	immediate = {
		set_country_flag = gray_diplomacy_engaged
		if = {
			limit = {
				any_owned_planet = {
					exists = leader
					leader = { is_same_value = event_target:gray_governor@root }
				}
			}
			random_owned_planet = {
				limit = {
					exists = leader
					leader = { is_same_value = event_target:gray_governor@root }
				}
				save_event_target_as = gray_governor_planet@root
			}
		}
		else = {
			capital_scope = { save_event_target_as = gray_governor_planet@root }
		}
	}

	after = {
		hidden_effect = { remove_country_flag = gray_diplomacy_engaged }
	}

	option = { # Warship
		name = graygoo.500.b
		response_text = graygoo.501.b.response
		allow = {
			hidden_trigger = { any_owned_leader = { is_same_value = event_target:gray_governor@root } } # To prevent exploit
		}
		hidden_effect = {
			create_gray_warship = yes
		}
	}
	option = { # Army
		name = graygoo.500.c
		response_text = graygoo.501.c.response
		allow = {
			hidden_trigger = { any_owned_leader = { is_same_value = event_target:gray_governor@root } }
		}
		hidden_effect = {
			create_gray_army = yes
		}
	}
	option = {
		name = graygoo.500.d
		default_hide_option = yes
	}
}

# Gray destroyed or disbanded
country_event = {
	id = gray_fix.110
	hide_window = yes

	is_triggered_only = yes

	trigger = {
		any_country = { is_country_type = gray_fix_country }
		OR = {
			AND = {
				has_country_flag = gray_army_active
				NOT = {
					any_owned_army = { army_type = gray_army }
				}
			}
			AND = {
				has_country_flag = gray_warship_active
				any_controlled_fleet = { is_same_value = event_target:gray_warship@root }
			}
			AND = {
				has_country_flag = gray_governor_active
				any_owned_leader = { is_same_value = event_target:gray_governor@root }
			}
		}
	}

	immediate = {
		remove_country_flag = gray_governor_active
		remove_country_flag = gray_warship_active
		remove_country_flag = gray_army_active
		set_country_flag = gray_reforming
		add_modifier = {
			modifier = "gray_scattered"
			days = 360
		}
		country_event = { id = gray_fix.111 }
		country_event = { id = gray_fix.112 days = 360 }
	}
}
